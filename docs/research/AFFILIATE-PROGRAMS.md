## Overview

### Revenue Potential
Affiliate programs typically pay:
- **Exchanges**: 20-50% of trading fees from referred users
- **Hardware Wallets**: 10-15% of sale price
- **Mining Hardware**: 3-10% of sale price (varies by manufacturer)

### Priority Order
1. High-volume exchanges (most user signups likely)
2. Hardware wallets (high trust, used in wallet section)
3. Mining hardware (specialized audience)

## Affiliate Link Bank Architecture

### Recommendation: Centralized Affiliate Link System

Yes, it makes sense to create an affiliate link bank. This provides:

1. **Single Source of Truth**: All affiliate links in one place
2. **Easy Updates**: Change affiliate codes without hunting through codebase
3. **Tracking**: Can add UTM parameters consistently
4. **Fallback**: Use non-affiliate links if program not yet joined

### Proposed Implementation

```typescript
// lib/affiliates.ts

export interface AffiliateLink {
  id: string
  name: string
  baseUrl: string
  affiliateUrl: string | null  // null if not yet signed up
  affiliateCode: string | null
  category: 'exchange' | 'hardware-wallet' | 'mining-hardware' | 'retailer'
  tier: 1 | 2 | 3
  isActive: boolean
  addedDate: string | null
  notes?: string
}

export const affiliateLinks: Record<string, AffiliateLink> = {
  // Tier 1 Exchanges
  binance: {
    id: 'binance',
    name: 'Binance',
    baseUrl: 'https://www.binance.com',
    affiliateUrl: null,  // Will be: https://www.binance.com/en/register?ref=XXXXX
    affiliateCode: null,
    category: 'exchange',
    tier: 1,
    isActive: false,
    addedDate: null,
  },
  coinbase: {
    id: 'coinbase',
    name: 'Coinbase',
    baseUrl: 'https://www.coinbase.com',
    affiliateUrl: null,
    affiliateCode: null,
    category: 'exchange',
    tier: 1,
    isActive: false,
    addedDate: null,
  },
  kraken: {
    id: 'kraken',
    name: 'Kraken',
    baseUrl: 'https://www.kraken.com',
    affiliateUrl: null,
    affiliateCode: null,
    category: 'exchange',
    tier: 1,
    isActive: false,
    addedDate: null,
  },

  // Hardware Wallets (already implemented)
  trezor: {
    id: 'trezor',
    name: 'Trezor',
    baseUrl: 'https://trezor.io',
    affiliateUrl: 'https://affil.trezor.io/aff_c?offer_id=133&aff_id=34561',
    affiliateCode: '34561',
    category: 'hardware-wallet',
    tier: 1,
    isActive: true,
    addedDate: '2026-01',
  },
  ledger: {
    id: 'ledger',
    name: 'Ledger',
    baseUrl: 'https://www.ledger.com',
    affiliateUrl: 'https://shop.ledger.com/?r=bbf4d7f32e72',
    affiliateCode: 'bbf4d7f32e72',
    category: 'hardware-wallet',
    tier: 1,
    isActive: true,
    addedDate: '2026-01',
  },
}

/**
 * Get the best available link for an affiliate
 * Returns affiliate link if active, otherwise returns base URL
 */
export function getAffiliateLink(id: string): string {
  const affiliate = affiliateLinks[id]
  if (!affiliate) {
    console.warn(`Unknown affiliate ID: ${id}`)
    return '#'
  }
  return affiliate.isActive && affiliate.affiliateUrl
    ? affiliate.affiliateUrl
    : affiliate.baseUrl
}

/**
 * Get all active affiliates by category
 */
export function getActiveAffiliates(category?: AffiliateLink['category']): AffiliateLink[] {
  return Object.values(affiliateLinks)
    .filter(a => a.isActive)
    .filter(a => !category || a.category === category)
}

/**
 * Check if an affiliate link is active
 */
export function isAffiliateActive(id: string): boolean {
  return affiliateLinks[id]?.isActive ?? false
}
```

### Usage Example

```typescript
// In any component
import { getAffiliateLink } from '@/lib/affiliates'

// Automatically uses affiliate link if available, otherwise base URL
<a href={getAffiliateLink('binance')}>Trade on Binance</a>
<a href={getAffiliateLink('trezor')}>Buy Trezor</a>
```

### Migration Path

1. Create `lib/affiliates.ts` with the structure above
2. Update `app/wallet/data/wallets.ts` to use `getAffiliateLink('trezor')` etc.
3. Update `app/buy/data/exchanges.ts` to use affiliate links
4. Update any hardcoded exchange links across the site
